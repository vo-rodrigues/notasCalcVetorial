\begin{definition}

        \labeleditem{(A2)} $x+y=y+x$ (propriedade comutativa); \label{defItem:comutativaSoma}
        \labeleditem{(A3)} $x+0=x$ (elemento neutro para adiĂ§ĂŁo); \label{defItem:neutroSoma}
        \labeleditem{(A4)} existe $u\in \mathbb R$ tal que $x+u=0$ (existĂŞncia de elemento inverso para adiĂ§ĂŁo); \label{defItem:inversoSoma} \label{defItem:ultimoSoma}

        \labeleditem{(M2)} $x\cdot 1=x$ e $1\cdot x=x$ (elemento neutro para multiplicaĂ§ĂŁo); \label{defItem:neutroMultiplicacao}
        \labeleditem{(M3)} $x(y+z)=x\cdot y+x\cdot z$ (propriedade distributiva); \label{defItem:distributiva}
        \labeleditem{(M4)} Se $x\neq 0$, existe $v\in \mathbb R$ tal que $x\cdot v=1$ e $v\cdot x=1$ (existĂŞncia de elemento inverso para multiplicaĂ§ĂŁo); \label{defItem:inversoMultiplicacao}
        \labeleditem{(M5)} $0\neq 1$ (nĂŁo trivialidade); \label{defItem:naoTrivialidade} \label{defItem:ultimoMultiplicacao}
        \labeleditem{(O1)} Se $x<y$ e $y<z$, entĂŁo $x<z$ (propriedade transitiva); \label{defItem:primeiraOrdem} \label{defItem:transitiva}
        \labeleditem{(O2)} $x \not< x$ (propriedade irreflexiva); \label{defItem:segundaOrdem} \label{defItem:irreflexiva}
        \labeleditem{(O3)} $x<y$ ou $x=y$ ou $y<x$ (propriedade de totalidade); \label{defItem:terceiraOrdem} \label{defItem:totalidade}
        \labeleditem{(OA)} Se $x<y$, entĂŁo $x+z<y+z$ (compatibilidade da adiĂ§ĂŁo com a ordem); \label{defItem:compatibilidadeSoma}
        \labeleditem{(OM)} Se $x<y$ e $0<z$, entĂŁo $x\cdot z<y\cdot z$ (compatibilidade da multiplicaĂ§ĂŁo com a ordem); \label{defItem:compatibilidadeMultiplicacao}
        \labeleditem{(S)} Todo subconjunto de $\mathbb R$ nĂŁo vazio e limitado superiormente tem um supremo (propriedade de completude). \label{defItem:supremo}
    \end{enumerate}

    Os elementos de $\mathbb R$ sĂŁo chamados de nĂşmeros reais.
\end{definition}

A Propriedade~\ref{defItem:supremo}, tambĂŠm conhecida como a propriedade do supremo, certamente requer explicaĂ§Ăľes adicionais.
DeixĂĄ-la-emos por Ăşltimo.
Solicitamos que o leitor nĂŁo se preocupe com ela atĂŠ que tal explicaĂ§ĂŁo seja concedida.

Iniciaremos discutindo as propriedades do grupo \ref{defItem:primeiraSoma}-\ref{defItem:ultimoSoma}.
Tais propriedades regem a adiĂ§ĂŁo de nĂşmeros reais isoladamente.

\begin{proposition}
    Seja $x$ um nĂşmero real.
    EntĂŁo existe um Ăşnico nĂşmero real $u$ tal que $x+u=0$.
\end{proposition}
\begin{proof}
    Tome um nĂşmero real $x$.
    Pela propriedade \ref{defItem:inversoSoma}, existe um nĂşmero real $u$ tal que $x+u=0$.
    Vejamos que sĂł existe um: ou seja, que se $u$ e $u'$ sĂŁo nĂşmeros reais tais que $x+u=0$ e $x+u'=0$, entĂŁo $u=u'$.

    De fato, se $u$ e $u'$ sĂŁo nĂşmeros reais tais que $x+u=0$ e $x+u'=0$, entĂŁo
\begin{align*}
    u &= u + 0 \tag{0 ĂŠ neutro da soma} \\
    &= u + (x + u) \tag{pela escolha de $u$} \\
    &= (u + x) + u' \tag{propriedade associativa} \\
    &= 0 + u' \tag{pela escolha de $u'$} \\
    &= u' \tag{0 ĂŠ neutro da soma}
\end{align*}
Assim, $u=u'$.
\end{proof}

Dado um nĂşmero real $x$, existe entĂŁo um Ăşnico nĂşmero real $u$ tal que $x+u=0$.
Tal nĂşmero $u$ ĂŠ completamente determinado por essa propriedade, e depende exclusivamente de $x$.

\begin{definition}
    Seja $x$ um nĂşmero real.
    O \emph{oposto}\index{oposto} de $x$, denotado por $-x$, ĂŠ o Ăşnico nĂşmero real tal que $x+(-x)=0$.
\end{definition}

Com isso, podemos provar algumas das chamadas \emph{regras de sinais}.

\begin{definition}
    Sejam $x$, $y$ nĂşmeros reais.
    Define-se a operaĂ§ĂŁo de subtraĂ§ĂŁo de nĂşmeros reais $-: \mathbb R\times \mathbb R\to \mathbb R$ por $x-y = x+(-y)$.
\end{definition}

\begin{proposition}

    \begin{enumerate}[label=(\alph*)]
        \item $-0=0$;
        \item $-(x+y)=(-x)+(-y)$;
        \item $-(-x)=x$.
    \end{enumerate}
\end{proposition}
\begin{proof}
    Para a propriedade $(a)$, temos que $0+0=0$. Como $-0$ ĂŠ o Ăşnico nĂşmero real tal que $0+(-0)=0$, entĂŁo $-0=0$.

    Para a propriedade $(b)$, temos que
    \begin{align*}
    (x+y)+((-x)+(-y)) &= [(x+y)+(-x)]+(-y) \tag{propriedade associativa} \\
    &= [(y+x)+(-x)]+(-y) \tag{propriedade comutativa} \\
    &= [y+(x+(-x))]+(-y) \tag{propriedade associativa} \\
    &= (y+0)+(-y) \tag{pela definiĂ§ĂŁo de oposto} \\
    &= y+(-y) \tag{0 ĂŠ neutro da soma} \\
    &= 0. \tag{pela definiĂ§ĂŁo de oposto}
    \end{align*}
    Como $-(x+y)$ ĂŠ o Ăşnico nĂşmero real tal que $(x+y)+(-(x+y))=0$, entĂŁo $-(x+y)=(-x)+(-y)$.

    Para a propriedade $(c)$, temos que $(-x)+x=0$.
    Como $-(-x)$ ĂŠ o Ăşnico nĂşmero real tal que $(-x)+(-(-x))=0$, entĂŁo $-(-x)=x$.
\end{proof}
