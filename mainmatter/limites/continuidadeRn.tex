\begin{definition}\index{projeĂ§ĂŁo}

\begin{proposition}\index{projeĂ§ĂŁo}
    Seja $n\geq 1$ e $i$ um inteiro com $1\leq i\leq n$.

\end{proposition}
\begin{proof}

    \begin{equation*}
        d(\pi_i(x), \pi_i(p)) = |x_i - p_i| \leq \sqrt{(x_1 - p_1)^2 + \cdots + (x_n - p_n)^2} = d(x, p).
    \end{equation*}
