\section{Regras bĂĄsicas de Continuidade em $\mathbb R^n$}


\begin{proposition}\index{projeĂ§ĂŁo}
    Seja $f: \mathbb R^n\rightarrow R$ uma projeĂ§ĂŁo, ou seja, uma funĂ§ĂŁo da forma $f(x_1, \dots, x_n) = x_i$ para algum $i \in \{1, \dots, n\}$.
    EntĂŁo $f$ ĂŠ contĂ­nua.
\end{proposition}
\begin{proof}
    Seja $p=(p_1, \dots, p_n) \in \mathbb R^n$ e $\epsilon>0$.

    Note que, dado qualquer $x=(x_1, \dots, x_n) \in \mathbb R^n$, temos que:
    \begin{equation*}
        d(f(x), f(p)) = |x_i - p_i| \leq \sqrt{(x_1 - p_1)^2 + \cdots + (x_n - p_n)^2} = d(x, p).
    \end{equation*}

    Assim, se $d(x, p)<\epsilon$, entĂŁo $d(f(x), f(p))<\epsilon$.
    Logo, tomando $\delta=\epsilon$, vemos que $f$ ĂŠ contĂ­nua em $p$.
\end{proof}

Como um primeiro exemplo, tratemos das projeĂ§Ăľes: as funĂ§Ăľes que $\pi_i: \mathbb R^n\rightarrow \mathbb R$ dadas por $\pi_i(x_1, \dots, x_n) = x_i$ para cada $i \in \{1, \dots, n\}$.

\begin{proposition}\index{projeĂ§ĂŁo}
    Seja $n\geq 1$ e $i$ um inteiro com $1\leq i\leq n$.
    
    Considere a $i$-ĂŠsima projeĂ§ĂŁo de $\mathbb R^n$ em $\mathbb R$, ou seja, a funĂ§ĂŁo $\pi_i: \mathbb R^n\rightarrow \mathbb R$ dada por $\pi_i(x_1, \dots, x_n) = x_i$.

    EntĂŁo $\pi_i$ ĂŠ contĂ­nua.
\end{proposition}
\begin{proof}
    Seja $p=(p_1, \dots, p_n) \in \mathbb R^n$ e $\epsilon>0$.

    Note que, dado qualquer $x=(x_1, \dots, x_n) \in \mathbb R^n$, temos que:
    \begin{equation*}
        d(\pi_i(x), \pi_i(p)) = |x_i - p_i| \leq \sqrt{(x_1 - p_1)^2 + \cdots + (x_n - p_n)^2} = d(x, p).
    \end{equation*}

    Assim, se $d(x, p)<\epsilon$, entĂŁo $d(\pi_i(x), \pi_i(p))<\epsilon$.
    Logo, tomando $\delta=\epsilon$, vemos que $\pi_i$ ĂŠ contĂ­nua em $p$.
\end{proof}

Outro exemplo de funĂ§ĂŁo contĂ­nua ĂŠ a funĂ§ĂŁo constante.

\begin{proposition}\index{funĂ§ĂŁo constante}
    Seja $f: \mathbb R^n\rightarrow \mathbb R$ uma funĂ§ĂŁo constante, ou seja, uma funĂ§ĂŁo da forma $f(x_1, \dots, x_n) = c$ para algum $c \in \mathbb R$.
    EntĂŁo $f$ ĂŠ contĂ­nua.
\end{proposition}
\begin{proof}
    Seja $p=(p_1, \dots, p_n) \in \mathbb R^n$ e $\epsilon>0$.

    Note que, dado qualquer $x=(x_1, \dots, x_n) \in \mathbb R^n$, temos que:
    \begin{equation*}
        d(f(x), f(p)) = |c - c| = 0 < \epsilon.
    \end{equation*}

    Assim, se $d(x, p)<\epsilon$, entĂŁo $d(f(x), f(p))<\epsilon$.
    Logo, tomando $\delta=\epsilon$, vemos que $f$ ĂŠ contĂ­nua em $p$.
\end{proof}

Lembremos que se $f, g: A \to \mathbb R$, podemos definir as funĂ§Ăľes $f+g$, $f-g$, $fg$ e $\frac{f}{g}$ (desde que $g(x)\neq 0$ para todo $x \in A$) como a seguir.

\begin{definition}
    Seja $A\subseteq \mathbb R^n$ um conjunto e $f, g: A \to \mathbb R$ funĂ§Ăľes.
    EntĂŁo, definimos as funĂ§Ăľes \emph{soma}\index{soma de funĂ§Ăľes}, \emph{diferenĂ§a}\index{diferenĂ§a de funĂ§Ăľes}, \emph{produto}\index{produto de funĂ§Ăľes} e \emph{quociente}\index{quociente de funĂ§Ăľes} por:
    \begin{align*}
        f+g &: A\rightarrow \mathbb R \\
        f-g &: A\rightarrow \mathbb R \\
        fg &: A\rightarrow \mathbb R \\
        \frac{f}{g} &: B\rightarrow \mathbb R,
    \end{align*}
    em que $B=\{x \in A : g(x)\neq 0\}$, e tais funĂ§Ăľes sĂŁo dadas por:
    \begin{align*}
        (f+g)(x) &= f(x) + g(x) \\
        (f-g)(x) &= f(x) - g(x) \\
        (fg)(x) &= f(x)g(x) \\
        \left(\frac{f}{g}\right)(x) &= \frac{f(x)}{g(x)}.
    \end{align*}
\end{definition}

\begin{proposition}\index{limite!operaĂ§Ăľes algĂŠbricas}
    Seja $A\subseteq \mathbb R^n$ um conjunto e $f, g: A \to \mathbb R$ funĂ§Ăľes.
    Seja $a \in A$ um ponto de acumulaĂ§ĂŁo de $A$.

    Sejam $L, S \in \mathbb R$ tais que $\lim_{x\to a} f(x) = L$ e $\lim_{x\to a} g(x) = S$.

    EntĂŁo:

    \begin{enumerate}[label=(\alph*)]
        \item $\lim_{x\to a} (f+g)(x) = L + S$;
        \item $\lim_{x\to a} (fg)(x) = LS$;
        \item $\lim_{x\to a} (f-g)(x) = L - S$.
    \end{enumerate}
\end{proposition}
\begin{proof}
    Provaremos primeiro o item (a).
    Seja dado $\epsilon>0$.

    Como $\lim_{x\to a} f(x) = L$, existe $\delta_f>0$ tal que, para todo $x \in A\setminus\{a\}$, se $d(x, a) < \delta_f$, entĂŁo $|f(x) - L| < \frac{\epsilon}{2}$.

    Analogamente, como $\lim_{x\to a} g(x) = S$, existe $\delta_g>0$ tal que, para todo $x \in A\setminus\{a\}$, se $d(x, a) < \delta_g$, entĂŁo $|g(x) - S| < \frac{\epsilon}{2}$.

    Seja $\delta = \min\{\delta_f, \delta_g\}$
    Agora, dado $x \in A\setminus\{a\}$ tal que $d(x, a) < \delta$, temos que:
    \begin{align*}
        |(f+g)(x) - (L + S)| &= |f(x) + g(x) - L - S| \\
        &= |(f(x) - L) + (g(x) - S)| \\
        &\leq |f(x) - L| + |g(x) - S| \\
        &< \frac{\epsilon}{2} + \frac{\epsilon}{2} = \epsilon.
    \end{align*}
      
    Logo, $\lim_{x\to a} (f+g)(x) = L + S$.

    Agora, provaremos o item (b).
    Seja dado $\epsilon>0$. Fixe um nĂşmero positivo $M$ qualquer.
    Como $\lim_{x\to a} f(x) = L$, existe $\delta_f>0$ tal que, para todo $x \in A\setminus\{a\}$, se $d(x, a) < \delta_f$, entĂŁo $|f(x) - L| < M$.

    Analogamente, como $\lim_{x\to a} g(x) = S$, existe $\delta_g>0$ tal que, para todo $x \in A\setminus\{a\}$, se $d(x, a) < \delta_g$, entĂŁo $|g(x) - S| < M$.

    Seja $\delta = \min\{\delta_f, \delta_g\}$

    Agora, dado $x \in A\setminus\{a\}$ tal que $d(x, a) < \delta$, temos que:

    \begin{align*}
        |(fg)(x) - LS| &= |f(x)g(x) - LS| \\
        &= |f(x)g(x) - f(x)S + f(x)S - LS| \\
        &= |f(x)(g(x) - S) + S(f(x) - L)| \\
        &\leq |f(x)||g(x) - S| + |S||f(x) - L| \\
        &< |f(x)|M + |S|M \\
        &= (|f(x)| + |S|)M.
    \end{align*}

    Notemos que, nessa hipĂłtese, $|f(x)|=|f(x) - L + L| \leq |f(x) - L| + |L| < M + |L|$.
    Logo, $|(fg)(x) - LS| < (M + |L| + |S|)M$.

    Tomando, de partida, $M$ tal que $M<1$ e $M < \frac{\epsilon}{1 + |L| + |S|}$, concluĂ­mos que $|(fg)(x) - LS| < (M + |L| + |S|)M<(1+|L|+|S|)\frac{\epsilon}{1 + |L| + |S|}=\epsilon$.

    Logo, $\lim_{x\to a} (fg)(x) = LS$.

    Para o item (c), note que $f-g = f + (-1)g$.
    Como a funĂ§ĂŁo constante $(-1)$ ĂŠ contĂ­nua, decorre dos itens anteriores que o limite de $f-g$ em $a$ ĂŠ $L +(-1)S = L - S$.
\end{proof}

\begin{corollary}\index{continuidade!operaĂ§Ăľes algĂŠbricas}
    Seja $A\subseteq \mathbb R^n$ um conjunto e $f, g: A \to \mathbb R$ funĂ§Ăľes e $a \in A$.

    Se $f, g$ sĂŁo contĂ­nuas em $a$, entĂŁo as funĂ§Ăľes $f+g$, $f-g$ e $fg$ sĂŁo contĂ­nuas em $a$.
\end{corollary}

\begin{proof}
    Se $a$ ĂŠ ponto isolado de $A$, entĂŁo $f, g, f+g, f-g$ e $fg$ sĂŁo contĂ­nuas em $a$.

    Caso contrĂĄrio, temos, da hipĂłtese de continuidade, que, $\lim_{x\to a} f(x) = f(a)$ e $\lim_{x\to a} g(x) = g(a)$.
    DaĂ­, da proposiĂ§ĂŁo anterior, temos que:
    \begin{align*}
        \lim_{x\to a} (f+g)(x) &= f(a) + g(a) = (f+g)(a) \\
        \lim_{x\to a} (f-g)(x) &= f(a) - g(a) = (f-g)(a) \\
        \lim_{x\to a} (fg)(x) &= f(a)g(a) = (fg)(a).
    \end{align*}
    Assim, $f+g$, $f-g$ e $fg$ sĂŁo contĂ­nuas em $a$.
\end{proof}

Com isso, temos alguns exemplos.

\begin{example}
    Seja $f:\mathbb R^3\rightarrow \mathbb R$ dada por $f(x, y, z) = 3x^2yz$.
    EntĂŁo $f$ ĂŠ contĂ­nua, pois $f$ ĂŠ o produto das funĂ§Ăľes contĂ­nuas $3$, $x$, $y$ e $z$.
    Ou seja, da funĂ§ĂŁo constante $3$, da projeĂ§ĂŁo na coordenada $x$, da projeĂ§ĂŁo na coordenada $y$ e da projeĂ§ĂŁo na coordenada $z$.
\end{example}

\begin{example}
    Seja $f:\mathbb R^2\rightarrow \mathbb R$ dada por $f(x, y) = 3x^2y + 2x + 5$.
    EntĂŁo $f$ ĂŠ contĂ­nua, pois ĂŠ soma de produtos de funĂ§Ăľes contĂ­nuas.
\end{example}

Para se efetivamente calcular limites, o seguinte resultado ĂŠ essencial.

\begin{proposition}
    Sejam $A\subseteq \mathbb R^n$, $f: A \to \mathbb R^m$ e $a$ ponto de acumulaĂ§ĂŁo de $A$.
    Suponha que exista uma funĂ§ĂŁo $g: A \to \mathbb R^m$ e uma bola aberta $B$ em torno de $a$ tal que, para todo $x \in B\cap A\setminus\{a\}$, temos que $f(x) = g(x)$.

    EntĂŁo, para qualquer que seja $L \in \mathbb R^m$, $\lim_{x\to a} f(x) = L$ se, e somente se, $\lim_{x\to a} g(x) = L$.
\end{proposition}
\begin{proof}
    Se $\lim_{x\to a} f(x) = L$, entĂŁo, para todo $\epsilon>0$, existe $\delta>0$ tal que, para todo $x \in A\setminus\{a\}$, se $d(x, a) < \delta$, entĂŁo $d(f(x), L) < \epsilon$.

    Seja $r>0$ tal que para todo $x \in B(a, r)\cap A\setminus\{a\}$, temos que $f(x) = g(x)$.
    Seja $\delta' = \min\{\delta, r\}$.
    EntĂŁo para todo $x \in A\setminus\{a\}$, se $d(x, a) < \delta'$, entĂŁo $d(g(x), L) = d(f(x), L) < \epsilon$.
    Logo, $\lim_{x\to a} g(x) = L$.

    A recĂ­proca ĂŠ anĂĄloga.
\end{proof}

Sobre funĂ§Ăľes compostas, temos o seguinte resultado.
\begin{proposition}\index{composiĂ§ĂŁo de funĂ§Ăľes}\index{limite!composiĂ§ĂŁo}
    Sejam $f: A \to \mathbb B$ e $g: B \to \mathbb R^k$, onde $A\subseteq \mathbb R^m$ e $B\subseteq \mathbb R^n$ sĂŁo conjuntos tais que $f(A)\subseteq B$.
    Seja $a \in \mathbb R^n$ um ponto de acumulaĂ§ĂŁo de $A$ e $L \in B$.

    Se $\lim_{x\to a} f(x) = L$ e $g$ ĂŠ contĂ­nua em $L$, entĂŁo $\lim_{x\to a} (g\circ f)(x) = g(L)$.
\end{proposition}

\begin{proof}
    Seja dado $\epsilon>0$.

    Como $g$ ĂŠ contĂ­nua em $L$, existe $\eta>0$ tal que, para todo $y \in B$, se $d(y, L) < \eta$, entĂŁo $d(g(y), g(L)) < \epsilon$.

    Como $\lim_{x\to a} f(x) = L$ e $g$ ĂŠ contĂ­nua em $L$, existe $\delta>0$ tal que, para todo $x \in A\setminus\{a\}$, se $d(x, a) < \delta$, entĂŁo $d(f(x), L) < \eta$.


    Agora, dado $x \in A\setminus\{a\}$ tal que $d(x, a) < \delta$, temos que $d(f(x), L) < \eta$, logo, $d((g\circ f)(x), g(L)) < \epsilon$.

    Assim, $\lim_{x\to a} (g\circ f)(x) = g(L)$.
\end{proof}

Como consequĂŞncia, temos os seguintes corolĂĄrios.

\begin{corollary}
    Sejam $f: A \to \mathbb B$ e $g: B \to \mathbb R^k$, onde $A\subseteq \mathbb R^m$ e $B\subseteq \mathbb R^n$ sĂŁo conjuntos tais que $f(A)\subseteq B$.
    Seja $a \in A$ um ponto de acumulaĂ§ĂŁo de $A$ e $L \in B$.

    Se $\lim_{x\to a} f(x) = L$ e $g$ ĂŠ contĂ­nua em $L$, entĂŁo $\lim_{x\to a} (g\circ f)(x) = g(L)$.
\end{corollary}

\begin{corollary}\index{funĂ§ĂŁo recĂ­proca}
    Sejam $f: A \to \mathbb R\setminus \{0\}$, com $A\subseteq \mathbb R^n$, e $a \in \mathbb R^n$ um ponto de acumulaĂ§ĂŁo de $A$.

    Se $\lim f(x) = L \neq 0$ e $f$ ĂŠ contĂ­nua em $a$, $\lim_{x\to a} \frac{1}{f(x)} = \frac{1}{L}$.
\end{corollary}

\begin{proof}
    Note que a funĂ§ĂŁo $g:\mathbb R\setminus\{0\}\rightarrow \mathbb R$ dada por $g(x) = \frac{1}{x}$ ĂŠ contĂ­nua em todo ponto de seu domĂ­nio.
\end{proof}

\begin{corollary}\index{continuidade!funĂ§ĂŁo recĂ­proca}
    Sejam $f: A \to \mathbb R\setminus \{0\}$, com $A\subseteq \mathbb R^n$, e $a \in A$.

    Se $f$ ĂŠ contĂ­nua em $a$, entĂŁo $\frac{1}{f}$ ĂŠ contĂ­nua em $a$.
\end{corollary}

Sobre composiĂ§ĂŁo de funĂ§Ăľes, temos tambĂŠm o seguinte resultado.
\begin{proposition}\index{limite!composiĂ§ĂŁo com curva}
    Sejam $f: A \rightarrow \mathbb R^n$, com $A\subseteq \mathbb R^m$, e $a$ ponto de acumulaĂ§ĂŁo de $A$.

    Suponha que exista $\lim_{x\to a} f(x) = L \in \mathbb R^n$.

    EntĂŁo, para todo $u: U\rightarrow A$, com $U\subseteq \mathbb R^k$ e $\alpha$ ponto de acumulaĂ§ĂŁo de $U$, se:

    \begin{enumerate}[label=(\alph*)]
        \item Existe uma bola aberta $B$ em torno de $\alpha$ tal que para todo $t \in B\cap U\setminus\{u\}$, temos que $u(t)\neq \alpha$.
        \item $\lim_{t\to \alpha} u(t) = a$.
    \end{enumerate}

    EntĂŁo $\lim_{t\to \alpha} (f\circ u)(t) = L$.
\end{proposition}

\begin{proof}
    Seja dado $\epsilon>0$.

    Como $\lim_{x\to a} f(x) = L$, existe $\delta>0$ tal que, para todo $x \in A\setminus\{a\}$, se $d(x, a) < \eta$, entĂŁo $d(f(x), L) < \epsilon$.

    Como $\lim_{t\to \alpha} u(t) = a$, existe $\delta_u>0$ tal que, para todo $t \in U\setminus\{\alpha\}$, se $d(t, \alpha) < \delta_u$, entĂŁo $d(u(t), a) < \eta$.

   AlĂŠm disso, existe $\delta'>0$ tal que para todo $t \in B(\alpha, \delta')$ com $t\neq \alpha$, temos que $u(t)\neq \alpha$.

    Seja $\delta = \min\{\delta_u, \delta'\}$.

    Agora, dado $t \in U\setminus\{\alpha\}$ tal que $d(t, \alpha) < \delta$, temos que $d(u(t), a) < \delta_u$ e $u(t)\neq \alpha$, logo, $d((f\circ u)(t), L) < \epsilon$.

    Assim, $\lim_{t\to \alpha} (f\circ u)(t) = L$.
\end{proof}

Com isso, temos o seguinte.

\begin{corollary}[Teste das curvas]\index{teste das curvas}\index{limite!teste das curvas}\index{curvas|seealso{curva, teste das curvas}}
    Sejam $f: A \rightarrow \mathbb R$, com $A\subseteq \mathbb R^m$, e $p$ ponto de acumulaĂ§ĂŁo de $A$.

    Sejam $u, v: I\rightarrow A$, com $I\subseteq \mathbb R$ um intervalo aberto contĂ­nuas em $\alpha \in I$ e $u(\alpha) = v(\alpha) = p$.

    Se existir o limite $L$ de $f$ em $p$, entĂŁo $L=\lim_{t\to \alpha} (f\circ u)(t) = \lim_{t\to \alpha} (f\circ v)(t)$.

    Em particular:

    \begin{itemize}
        \item Se $f\circ u$ ou $f\circ v$ nĂŁo possuĂ­rem limites em $\alpha$, entĂŁo $f$ nĂŁo possui limite em $p$.
        \item Se $f\circ u$ e $f\circ v$ possuĂ­rem limites em $\alpha$, mas $\lim_{t\to \alpha} (f\circ u)(t) \neq \lim_{t\to \alpha} (f\circ v)(t)$, entĂŁo $f$ nĂŁo possui limite em $p$.
        \item Se $f\circ u$ e $f\circ v$ possuĂ­rem limites em $\alpha$, e ambos forem iguais, o teste ĂŠ inconclusivo, mas outras curvas podem ser testadas.
    \end{itemize}
\end{corollary}
